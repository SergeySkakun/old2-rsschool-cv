Snail Sort 
snail = function (array) {
  let result = [];

  while (array.length > 1) {
    let temp = [];
    let length = array.length;
    result = result.concat(array.shift());

    for (let i = array[0].length - 1; i >= 0; i--) {
      let newLine = [];
      for (let l = 0; l < length; l++) {
        newLine.push(array[l][i]);
      }
      temp.push(newLine);
    }

    array = temp;
  }

  result = result.concat(array.shift());
  return result;
};
